LightField computeLightField(SceneData scene, HandPose hand) {
LightField lf;
#pragma omp parallel for collapse(2)
for (int x = 0; x < 512; x++) {
for (int y = 0; y < 512; y++) {
Ray ray = computeEyeRay(x, y, hand.position);
Intersection hit = rayMarch(scene, ray);
for (int depth = 0; depth < 256; depth++) {
lf.depth_map[depth][x][y] = hit.depth < depth_plane[depth] ? 1.0 : 0.0;
}
lf.color_map[x][y] = hit.color;
}
}
return lf;
}
