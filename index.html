// Trajectory planner - minimum jerk with lookahead collision avoidance
Trajectory computeMinimumJerk(Waypoint start, Waypoint end, float max_vel, float max_acc) {
Trajectory traj;
float T = computeOptimalTime(start, end, max_vel, max_acc);
for (float t = 0; t <= T; t += SAMPLE_TIME) {
float s = 10*pow(t/T,3) - 15*pow(t/T,4) + 6*pow(t/T,5); // quintic blend
traj.q[t] = start.q + s * (end.q - start.q);
}
return traj;
}

// Force/impedance control loop - 4kHz update rate
void impedanceControl(RobotState* state, Vector6f desired_force, Matrix6f stiffness, Matrix6f damping) {
Vector6f force_error = desired_force - state->fts_sensor;
Vector6f velocity_cmd = stiffness.inverse() * (force_error - damping * state->cart_vel);
state->dq_cmd = jacobian_pseudo_inverse(state->jacobian) * velocity_cmd;
}
